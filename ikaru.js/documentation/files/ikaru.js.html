                            if (typeof style === "string") return _getComputedStyle(element, cssNameProperty(style));
                        
                            //get array of elements
                            if (_isArray(style)) {
                              var css = {};
                              for (i in style) {
                                css[style[i]] = _getComputedStyle(element, cssNameProperty(style[i]));
                              }
                              return css;
                            }
                        
                            if (_isLiteralObject(style)) {
                              //set csses
                              for (i in style) {
                                element.style[cssNameProperty(i)] = style[i];
                              }
                              return style;
                            }
                            return false;
                          };
                        
                        
                          /**
                           * Gets css classes
                           *
                           * @method classess
                           * @param element {Object} HTML Element
                           * @param style {Object} json object with css style
                           * @return {Array} Classnames
                           */
                          I.classess = function(element) {
                            _checkDomRequired(element);
                        
                            var attribute = I.attribute(element, 'class');
                            if (!attribute) return [];
                            attribute = attribute.split(' ');
                            var classNames = [];
                            for (var i in attribute) {
                              if (attribute[i] === '') continue;
                              classNames.push(attribute[i]);
                            }
                            return classNames;
                          };
                        
                          /**
                           * Check if HTML Element has a Class
                           *
                           * @method hasClass
                           * @param element {Object} HTML Element
                           * @param className {String} json object with css style
                           * @return {Array} Classnames
                           */
                          I.hasClass = function(element, className) {
                            _checkDomRequired(element);
                        
                            if (_isString(className)) {
                              return _indexOf(I.classess(element), className) > -1 ? true : false;
                            } else if (_isArray(className)) {
                              var elementClasses = I.classess(element);
                              for (var i in className) {
                                if (_indexOf(className[i], elementClasses) === -1) {
                                  return false;
                                }
                              }
                              return true;
                            } else {
                              return false;
                            }
                        
                            return false;
                          };
                        
                        
                          /**
                           * Add Class to HTML object
                           *
                           * @method addClass
                           * @param element {Object} HTML Element
                           * @param className {String} string with className
                           * @return {Array} Classnames
                           */
                          I.addClass = function(element, className) {
                            if (element === undefined) _error("first parameter cannot be undefined");
                        
                            if (_isIterable(element)) {
                              _each(element, function(e) {
                                I.addClass(e, className);
                              });
                              return Dom;
                            }
                        
                            _checkDomRequired(element);
                        
                            if (_isArray(className)) {
                              for (var i in className) {
                                I.addClass(element, className[i]);
                              }
                              return Dom;
                            }
                        
                            var classes = I.classess(element);
                        
                            if (_indexOf(classes, className) === -1) {
                              classes.push(className);
                            }
                            classes = classes.join(' ');
                            return I.attribute(element, {
                              class: classes
                            });
                          };
                        
                          /**
                           * Remove Class
                           *
                           * @method removeClass
                           * @param element {Object} HTML Element
                           * @param className {String} string with className
                           * @return {Array} Classnames
                           */
                          I.removeClass = function(element, className) {
                            if (element === undefined) _error("first parameter cannot be undefined");
                        
                            if (_isIterable(element)) {
                              _each(element, function(e) {
                                I.removeClass(e, className);
                              });
                              return Dom;
                            }
                            if (!I.isDom(element)) _error("I.removeClass" + element + " is not a DOMNode object");
                        
                            if (!className) {
                              return I.attribute(element, {
                                class: null
                              });
                            }
                        
                            var classes = I.classess(element);
                            var i = _indexOf(classes, className);
                        
                            if (i === -1) return;
                        
                            classes.splice(i, 1);
                            return I.attribute(element, {
                              class: classes.join(' ')
                            });
                          };
                        
                        
                        
                        
                          /**
                           * HTML MANIPULATION
                           *
                           * @class HTML
                           */
                        
                        
                          /**
                           * Set or Get HTML text from  HTML element
                           *
                           * @method html
                           *
                           * @param element {String} HTML content to insert in new element
                           * @param string {String} HTML content to insert in new element (optional)
                           *
                           * @return {Object} String or HTML content from element
                           */
                          I.html = function(element, string) {
                            _checkDomRequired(element);
                        
                            if (_isString(string)) {
                              element.innerHTML = string;
                              return element;
                            }
                        
                            return element.innerHTML;
                          };
                        
                          /**
                           * Create a div element with specific HTML content
                           *
                           * @method createHtml
                           * @param html {String} HTML content to insert in new element
                           * @return {Object} New HTML ELement with specific html content
                           */
                          I.createHtml = function(html) {
                            var div = document.createElement('div');
                            var elem = I.html(div, html);
                        
                            return elem;
                          };
                        
                          /**
                           * Create a DOMNode copy
                           *
                           * @method copy
                           * @param element {Object} HTML element
                           * @return {Object} Cloned HTML ELement
                           */
                          I.copy = function(element) {
                            _checkDomRequired(element);
                            return element.cloneNode(true);
                          };
                        
                        
                          /**
                           * Set or get Text to HTML element
                           *
                           * @method text
                           * @param element {Object} HTML element
                           * @param string {String} Text to set in HTML Element (optional)
                           * @optional
                           *
                           * @return {Object} Cloned HTML ELement
                           */
                          I.text = function(element, string) {
                            _checkDomRequired(element);
                        
                            if (_isString(string)) {
                        
                              if (element.innerText) {
                                element.innerText = string;
                              } else {
                                element.textContent = string;
                              }
                              return string;
                            }
                        
                            if (element.innerText) return element.innerText;
                            return element.textContent;
                        
                          };
                        
                          /**
                           * Insert new HTML element to the END from HTML Element
                           *
                           * @method append
                           * @param element {Object} HTML element
                           * @param html {Mixed} If Text, create new element, If HTML Element, append to element
                           *
                           * @return {Object} HTML element appended
                           */
                          I.append = function(element, html) {
                            _checkDomRequired(element);
                        
                            if (_isString(html)) html = I.createHtml(html);
                        
                            element.appendChild(html);
                            return element;
                          };
                        
                        
                          /**
                           * Insert html content to the BEGIN from HTML Element
                           *
                           * @method prepend
                           * @param element {Object} HTML element
                           * @param html {Mixed} If Text, create new element, If HTML Element, prepend to element
                           *
                           * @return {Object} HTML element prepended
                           */
                          I.prepend = function(element, html) {
                            _checkDomRequired(element);
                        
                            if (_isString(html)) html = I.create(html);
                        
                            element.insertBefore(html, element.firstChild);
                            return element;
                          };
                        
                        
                          /**
                           * Insert html content AFTER from HTML Element
                           *
                           * @method after
                           * @param element {Object} HTML element
                           * @param html {Mixed} If Text, create new element, If HTML Element, insert after to element
                           *
                           * @return {Object} HTML element
                           */
                          I.after = function(element, html) {
                            _checkDomRequired(element);
                        
                            if (_isString(html)) {
                              html = I.create(html);
                            }
                        
                            element.parentNode.insertBefore(html, element.nextSibling);
                            return element;
                          };
                        
                        
                          /**
                           * Insert html content BEFORE from HTML Element
                           *
                           * @method after
                           * @param element {Object} HTML element
                           * @param html {Mixed} If Text, create new element, If HTML Element, insert before to element
                           *
                           * @return {Object} HTML element
                           */
                          I.before = function(element, html) {
                            _checkDomRequired(element);
                        
                            if (_isString(html)) {
                              html = I.create(html);
                            }
                        
                            element.insertBefore(html, element);
                            return html;
                          };
                        
                          /**
                           * Replace HTML text or HTML Element on element
                           *
                           * @method replace
                           * @param element {Object} HTML element
                           * @param html {Mixed} If Text, create new element, If HTML Element, replace child html with element
                           *
                           * @return {Object} HTML element replaced
                           */
                          I.replace = function(element, html) {
                            _checkDomRequired(element);
                        
                            if (_isString(html)) html = I.create(html);
                        
                            element.parentNode.replaceChild(html, element);
                            return element;
                          };
                        
                        
                          /**
                           * Remove DOM element
                           *
                           * @method remove
                           * @param element {Object} HTML element
                           *
                           */
                          I.remove = function(element) {
                            _checkDomRequired(element);
                        
                            return element.parentNode.removeChild(element);
                          };
                        
                        
                          /**
                           * MAIN EVENTS
                           *
                           * @class EVENTS
                           */
                        
                        
                        
                          /**
                           * Executed when PAGE is loaded
                           *
                           * @method loaded
                           * @param handler {function} function
                           *
                           */
                          I.loaded = function(handler) {
                            if (_isDomLoaded !== false) {
                              handler.call(null, _isDomLoaded);
                              return Dom;
                            }
                            _domLoadedHandlers.push(handler);
                            return Dom;
                          };
                        
                          // executed when document will be ready
                          I.ready = function(handler) {
                            if (_isDomReady !== false) {
                              handler.call(null, _isDomReady);
                              return Dom;
                            }
                            _domReadyHandlers.push(handler);
                            return Dom;
                          };
                        
                          function _onDOMLoaded(e) {
                        
                            var event = new I.Event(e);
                            _isDomLoaded = event;
                        
                            _each(_domLoadedHandlers, function(fn) {
                              fn.call(null, event);
                            });
                          }
                        
                          //on load
                          if (window.onload !== null) _domLoadedHandlers.push(window.onload);
                        
                          window.onload = _onDOMLoaded;
                        
                          //on ready
                          function _onDOMReady(e) {
                            //add most used selectors
                            I.body = I.tag('body')[0];
                            I.head = I.tag('head')[0];
                        
                            var event = new I.Event(e);
                            _isDomReady = event;
                        
                            _each(_domReadyHandlers, function(fn) {
                              fn.call(null, event);
                            });
                          }
                          if (addListener === 'attachEvent') { //shitty browsers
                            document[addListener]('onreadystatechange', function(e) {
                              if (document.readyState === 'complete') {
                                document[removeListener]('onreadystatechange', arguments.callee);
                                _onDOMReady(e);
                              }
                            });
                          } else { //ecma compatible browsers
                            document[addListener]('DOMContentLoaded', function(e) {
                              document[removeListener]('DOMContentLoaded', arguments.callee, false);
                              _onDOMReady(e);
                            }, false);
                          }
                        
                        
                          //export dom
                          window.Dom = Dom;
                        })(this, document);
